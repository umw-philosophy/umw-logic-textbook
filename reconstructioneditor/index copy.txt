  <script>
    const TAGS = {
      UNTAGGED: 'untagged',
      OMIT: 'omit',
      INDEPENDENT: 'independent',
      DEPENDENT: 'dependent',
      SUB: 'sub',
      MAIN: 'main'
    };

    const TAG_LABELS = {
      [TAGS.UNTAGGED]: 'Untagged',
      [TAGS.OMIT]: 'Omit from Reconstruction',
      [TAGS.INDEPENDENT]: 'Independent Premise',
      [TAGS.DEPENDENT]: 'Dependent Premise',
      [TAGS.SUB]: 'Sub-Conclusion',
      [TAGS.MAIN]: 'Main Conclusion'
    };

    const EXAMPLE_TEXT = 'The student newspaper has published three investigations this semester. Those investigations revealed unsafe housing conditions and errors in the student-fee budget. Reporting that uncovers safety problems and budget errors gives students information they need. The paper also trains new writers and editors. A project that informs students and develops student journalists is worth protecting. So the university has good reason to keep funding the student newspaper.';

    let units = [];
    let supportNotes = {};
    let nextId = 1;
    let activeSupportId = null;
    let lastFocus = null;
    let pendingCopy = false;

    const telemetry = {
      startedAt: Date.now(),
      tagActions: 0,
      revisionActions: 0,
      implicitAdded: 0
    };

    const $ = (id) => document.getElementById(id);

    function escapeHtml(value) {
      return String(value).replace(/[&<>"']/g, (char) => ({
        '&': '&amp;',
        '<': '&lt;',
        '>': '&gt;',
        '"': '&quot;',
        "'": '&#39;'
      }[char]));
    }

    function createUnit(text, options = {}) {
      return {
        id: `u${nextId++}`,
        text: text.trim(),
        tag: options.tag || TAGS.UNTAGGED,
        implicit: Boolean(options.implicit),
        createdAt: Date.now(),
        updatedAt: Date.now()
      };
    }

    function splitIntoStatements(text) {
      const normalized = text
        .replace(/\r\n/g, '\n')
        .replace(/\s+\n/g, '\n')
        .replace(/\n\s+/g, '\n')
        .trim();

      if (!normalized) return [];

      const protectedText = normalized
        .replace(/\b(U\.S|U\.K|D\.C|e\.g|i\.e|Mr|Mrs|Ms|Dr|Prof)\./g, (m) => m.replace(/\./g, '∯'));

      const parts = protectedText
        .split(/(?<=[.!?])\s+(?=[A-Z0-9"“])/)
        .flatMap((part) => part.split(/\n+/))
        .map((part) => part.replace(/∯/g, '.').trim())
        .filter(Boolean);

      return parts.length ? parts : [normalized];
    }

    function lineType(tag) {
      if (tag === TAGS.INDEPENDENT || tag === TAGS.DEPENDENT) return 'premise';
      if (tag === TAGS.SUB) return 'sub';
      if (tag === TAGS.MAIN) return 'main';
      return 'none';
    }

    function getStandardItems() {
      let premiseCount = 0;
      let subCount = 0;
      return units
        .filter((unit) => lineType(unit.tag) !== 'none')
        .map((unit) => {
          const type = lineType(unit.tag);
          let label = '';
          if (type === 'premise') label = `P${++premiseCount}`;
          if (type === 'sub') label = `C${++subCount}`;
          if (type === 'main') label = 'C';
          return { ...unit, type, label };
        });
    }

    function getLabelMap() {
      return Object.fromEntries(getStandardItems().map((item) => [item.id, item.label]));
    }

    function getAvailableSupportItems(targetId) {
      const items = getStandardItems();
      const targetIndex = items.findIndex((item) => item.id === targetId);
      if (targetIndex < 0) return [];
      return items.slice(0, targetIndex).filter((item) => item.type === 'premise' || item.type === 'sub');
    }

    function formatSupportNote(unitId) {
      const note = supportNotes[unitId];
      if (!note || !note.supportIds || !note.supportIds.length || !note.supportKind) {
        return { text: 'Needs support note revision', valid: false };
      }

      const labels = getLabelMap();
      const supportLabels = note.supportIds.map((id) => labels[id]).filter(Boolean);
      if (supportLabels.length !== note.supportIds.length || !supportLabels.length) {
        return { text: 'Needs support note revision', valid: false };
      }

      const joined = joinLabels(supportLabels);
      const kind = note.supportKind === 'dependent' ? 'dependent' : 'independent';
      const detail = note.detail ? `; ${note.detail}` : '';
      return { text: `(from ${joined}, ${kind}${detail})`, valid: true };
    }

    function joinLabels(labels) {
      if (labels.length <= 1) return labels[0] || '';
      if (labels.length === 2) return `${labels[0]} and ${labels[1]}`;
      return `${labels.slice(0, -1).join(', ')}, and ${labels[labels.length - 1]}`;
    }

    function render() {
      renderCards();
      renderStandardForm();
      updateStats();
      updateSourceStatus();
    }

    function renderCards() {
      const list = $('statement-list');
      if (!units.length) {
        list.innerHTML = '<div class="empty-state">Prepared statement cards will appear here.</div>';
        return;
      }

      list.innerHTML = units.map((unit, index) => {
        const tagLabel = TAG_LABELS[unit.tag];
        const implicitBadge = unit.implicit ? '<span class="badge badge-implicit">Implicit</span>' : '';
        const tagButtons = [
          [TAGS.OMIT, 'Omit'],
          [TAGS.INDEPENDENT, 'Independent Premise'],
          [TAGS.DEPENDENT, 'Dependent Premise'],
          [TAGS.SUB, 'Sub-Conclusion'],
          [TAGS.MAIN, 'Main Conclusion']
        ].map(([tag, label]) => `
          <button class="tag-btn" type="button" data-action="tag" data-id="${unit.id}" data-tag="${tag}" aria-pressed="${unit.tag === tag ? 'true' : 'false'}">${label}</button>
        `).join('');

        return `
          <article class="statement-card" data-tag="${unit.tag}" aria-labelledby="title-${unit.id}">
            <div class="card-head">
              <div id="title-${unit.id}" class="card-title">Statement ${index + 1}</div>
              <div>
                <span class="badge ${unit.tag === TAGS.UNTAGGED ? 'badge-warning' : ''}">${tagLabel}</span>
                ${implicitBadge}
              </div>
            </div>
            <label for="text-${unit.id}">Statement text</label>
            <textarea id="text-${unit.id}" class="statement-text" data-action="edit" data-id="${unit.id}">${escapeHtml(unit.text)}</textarea>
            <div class="tag-row" role="group" aria-label="Tag statement ${index + 1}">
              ${tagButtons}
            </div>
            <div class="card-tools">
              <button class="btn btn-small btn-secondary" type="button" data-action="split" data-id="${unit.id}">Split at Cursor</button>
              <button class="btn btn-small btn-secondary" type="button" data-action="merge-prev" data-id="${unit.id}" ${index === 0 ? 'disabled' : ''}>Merge with Previous</button>
              <button class="btn btn-small btn-secondary" type="button" data-action="merge-next" data-id="${unit.id}" ${index === units.length - 1 ? 'disabled' : ''}>Merge with Next</button>
              <button class="btn btn-small btn-danger" type="button" data-action="delete" data-id="${unit.id}">Delete</button>
            </div>
          </article>
        `;
      }).join('');
    }

    function renderStandardForm() {
      const container = $('standard-output');
      const items = getStandardItems();
      if (!items.length) {
        container.innerHTML = '<div class="empty-state">No premises or conclusions tagged yet.</div>';
        return;
      }

      container.innerHTML = `
        <ol class="standard-form" aria-label="Generated standard form reconstruction">
          ${items.map((item) => {
            const support = item.type === 'sub' || item.type === 'main' ? formatSupportNote(item.id) : null;
            const implicit = item.implicit ? ' <span class="badge badge-implicit">Implicit</span>' : '';
            const supportText = support
              ? `<span class="support-note ${support.valid ? '' : 'needs-revision'}">${escapeHtml(support.text)}</span>`
              : '';
            return `
              <li>
                <span class="line-label">${item.label}:</span>
                ${implicit}
                <span>${escapeHtml(item.text)}</span>
                ${supportText}
              </li>
            `;
          }).join('')}
        </ol>
      `;
    }

    function updateSourceStatus() {
      const untagged = units.filter((unit) => unit.tag === TAGS.UNTAGGED).length;
      const omitted = units.filter((unit) => unit.tag === TAGS.OMIT).length;
      const used = getStandardItems().length;
      const status = $('source-status');
      status.className = `status ${untagged ? 'warn' : 'good'}`;
      status.textContent = units.length
        ? `${units.length} cards prepared. ${used} in standard form. ${omitted} omitted. ${untagged} untagged.`
        : 'No statement cards yet.';
    }

    function updateStats() {
      $('time-stat').textContent = `Time: ${formatDuration(Date.now() - telemetry.startedAt)}`;
      $('action-stat').textContent = `Actions: ${telemetry.tagActions}`;
      $('revision-stat').textContent = `Revisions: ${telemetry.revisionActions}`;
      $('implicit-stat').textContent = `Implicit: ${telemetry.implicitAdded}`;
    }

    function formatDuration(ms) {
      const total = Math.floor(ms / 1000);
      const minutes = Math.floor(total / 60);
      const seconds = String(total % 60).padStart(2, '0');
      return `${minutes}m ${seconds}s`;
    }

    function announce(message) {
      $('live-region').textContent = message;
    }

    function prepareStatements() {
      const text = $('source-input').value.trim();
      if (!text) {
        $('source-status').className = 'status warn';
        $('source-status').textContent = 'Paste a passage before preparing statements.';
        return;
      }
      if (units.length && !window.confirm('Replace the current statement cards with new cards from the source text?')) return;
      units = splitIntoStatements(text).map((part) => createUnit(part));
      supportNotes = {};
      telemetry.revisionActions++;
      render();
      announce(`${units.length} statement cards prepared.`);
    }

    function loadExample() {
      $('source-input').value = EXAMPLE_TEXT;
      $('source-input').focus();
      announce('Example passage loaded.');
    }

    function clearAll() {
      if (!units.length && !$('source-input').value.trim()) return;
      if (!window.confirm('Start over and clear the current reconstruction?')) return;
      units = [];
      supportNotes = {};
      $('source-input').value = '';
      telemetry.revisionActions++;
      render();
      $('copy-status').className = 'status';
      $('copy-status').textContent = 'Nothing copied yet.';
      announce('Editor cleared.');
    }

    function findUnit(id) {
      return units.find((unit) => unit.id === id);
    }

    function unitIndex(id) {
      return units.findIndex((unit) => unit.id === id);
    }

    function updateText(id, text) {
      const unit = findUnit(id);
      if (!unit) return;
      unit.text = text;
      unit.updatedAt = Date.now();
      renderStandardForm();
    }

    function tagUnit(id, tag) {
      const unit = findUnit(id);
      if (!unit || unit.tag === tag) return;
      if (tag === TAGS.MAIN) {
        units.forEach((other) => {
          if (other.id !== id && other.tag === TAGS.MAIN) {
            other.tag = TAGS.UNTAGGED;
            other.updatedAt = Date.now();
            delete supportNotes[other.id];
          }
        });
      }
      unit.tag = tag;
      unit.updatedAt = Date.now();
      telemetry.tagActions++;

      if (tag !== TAGS.SUB && tag !== TAGS.MAIN) {
        delete supportNotes[id];
      }

      render();

      if (tag === TAGS.SUB || tag === TAGS.MAIN) {
        openSupportModal(id);
      } else {
        announce(`Statement tagged as ${TAG_LABELS[tag]}.`);
      }
    }

    function splitUnit(id) {
      const index = unitIndex(id);
      const unit = units[index];
      const textarea = $(`text-${id}`);
      if (!unit || !textarea) return;
      const cursor = textarea.selectionStart;
      const text = textarea.value;
      if (cursor <= 0 || cursor >= text.length) {
        announce('Place the cursor inside the statement text before splitting.');
        textarea.focus();
        return;
      }
      const before = text.slice(0, cursor).trim();
      const after = text.slice(cursor).trim();
      if (!before || !after) {
        announce('Split point must leave text on both sides.');
        textarea.focus();
        return;
      }
      unit.text = before;
      unit.updatedAt = Date.now();
      const newUnit = createUnit(after);
      units.splice(index + 1, 0, newUnit);
      telemetry.revisionActions++;
      render();
      setTimeout(() => $(`text-${newUnit.id}`)?.focus(), 0);
      announce('Statement split.');
    }

    function mergeUnit(id, direction) {
      const index = unitIndex(id);
      if (index < 0) return;
      const otherIndex = direction === 'prev' ? index - 1 : index + 1;
      if (otherIndex < 0 || otherIndex >= units.length) return;
      const firstIndex = Math.min(index, otherIndex);
      const secondIndex = Math.max(index, otherIndex);
      const first = units[firstIndex];
      const second = units[secondIndex];
      first.text = `${first.text.trim()} ${second.text.trim()}`.trim();
      first.updatedAt = Date.now();
      if (second.tag === TAGS.SUB || second.tag === TAGS.MAIN) delete supportNotes[second.id];
      Object.values(supportNotes).forEach((note) => {
        note.supportIds = note.supportIds.map((sid) => sid === second.id ? first.id : sid);
        note.supportIds = [...new Set(note.supportIds)];
      });
      units.splice(secondIndex, 1);
      telemetry.revisionActions++;
      render();
      announce('Statements merged.');
    }

    function deleteUnit(id) {
      const index = unitIndex(id);
      if (index < 0) return;
      const unit = units[index];
      if (!window.confirm('Delete this statement card?')) return;
      units.splice(index, 1);
      delete supportNotes[id];
      Object.values(supportNotes).forEach((note) => {
        note.supportIds = note.supportIds.filter((sid) => sid !== id);
      });
      telemetry.revisionActions++;
      render();
      announce(`${unit.implicit ? 'Implicit premise' : 'Statement'} deleted.`);
    }

    function openImplicitModal() {
      lastFocus = document.activeElement;
      $('implicit-text').value = '';
      hideError('implicit-error');
      showModal('implicit-backdrop');
      setTimeout(() => $('implicit-text').focus(), 0);
    }

    function saveImplicit() {
      const text = $('implicit-text').value.trim();
      if (!text) {
        showError('implicit-error', 'Enter an implicit premise before adding it.');
        return;
      }
      units.push(createUnit(text, { tag: TAGS.DEPENDENT, implicit: true }));
      telemetry.implicitAdded++;
      telemetry.revisionActions++;
      closeModal('implicit-backdrop');
      render();
      announce('Implicit premise added.');
    }

    function openSupportModal(id) {
      const unit = findUnit(id);
      if (!unit) return;
      activeSupportId = id;
      lastFocus = document.activeElement;
      hideError('support-error');
      $('support-target').textContent = unit.text;

      const existing = supportNotes[id] || { supportIds: [], supportKind: '', detail: '' };
      const options = getAvailableSupportItems(id);
      $('support-options').innerHTML = options.length ? options.map((item) => `
        <label class="support-option">
          <input type="checkbox" value="${item.id}" ${existing.supportIds.includes(item.id) ? 'checked' : ''}>
          <span><strong>${item.label}:</strong> ${escapeHtml(item.text)}</span>
        </label>
      `).join('') : '<p class="note">No earlier premises or sub-conclusions are available. Add supporting claims first, then revise this note.</p>';

      document.querySelectorAll('input[name="support-kind"]').forEach((radio) => {
        radio.checked = radio.value === existing.supportKind;
      });
      $('support-detail').value = existing.detail || '';
      showModal('support-backdrop');
      setTimeout(() => {
        const first = $('support-options').querySelector('input') || document.querySelector('input[name="support-kind"]');
        first?.focus();
      }, 0);
    }

    function saveSupport() {
      if (!activeSupportId) return;
      const supportIds = [...$('support-options').querySelectorAll('input[type="checkbox"]:checked')].map((input) => input.value);
      const relation = document.querySelector('input[name="support-kind"]:checked')?.value || '';
      if (!supportIds.length) {
        showError('support-error', 'Select at least one supporting statement.');
        return;
      }
      if (!relation) {
        showError('support-error', 'Select independent or dependent support.');
        return;
      }
      supportNotes[activeSupportId] = {
        supportIds,
        supportKind: relation,
        detail: $('support-detail').value.trim()
      };
      telemetry.revisionActions++;
      const savedId = activeSupportId;
      activeSupportId = null;
      closeModal('support-backdrop');
      render();
      const label = getLabelMap()[savedId] || 'Conclusion';
      announce(`Support note saved for ${label}.`);
    }

    function buildSubmissionText() {
      const items = getStandardItems();
      const lines = items.map((item) => {
        const implicit = item.implicit ? ' [Implicit]' : '';
        const support = item.type === 'sub' || item.type === 'main' ? ` ${formatSupportNote(item.id).text}` : '';
        return `${item.label}:${implicit} ${item.text}${support}`;
      });
      const receipt = `--- SESSION DATA: Time: ${formatDuration(Date.now() - telemetry.startedAt)} | Actions: ${telemetry.tagActions} | Revisions: ${telemetry.revisionActions} | Implicit Added: ${telemetry.implicitAdded} ---`;
      return `${lines.join('\n')}\n\n${receipt}`;
    }

    function requestCopy() {
      if (!getStandardItems().length) {
        $('copy-status').className = 'status warn';
        $('copy-status').textContent = 'Tag at least one premise or conclusion before copying.';
        return;
      }
      const untagged = units.filter((unit) => unit.tag === TAGS.UNTAGGED).length;
      if (untagged) {
        pendingCopy = true;
        lastFocus = document.activeElement;
        $('confirm-summary').textContent = `${untagged} statement${untagged === 1 ? ' is' : 's are'} still untagged.`;
        showModal('confirm-backdrop');
        setTimeout(() => $('confirm-copy-btn').focus(), 0);
        return;
      }
      copySubmission();
    }

    async function copySubmission() {
      pendingCopy = false;
      if ($('confirm-backdrop').classList.contains('active')) {
        closeModal('confirm-backdrop');
      }
      const text = buildSubmissionText();
      try {
        await navigator.clipboard.writeText(text);
        $('copy-status').className = 'status good';
        $('copy-status').textContent = 'Standard form copied with process receipt.';
        announce('Standard form copied.');
      } catch (error) {
        fallbackCopy(text);
      }
    }

    function fallbackCopy(text) {
      const helper = document.createElement('textarea');
      helper.value = text;
      helper.setAttribute('readonly', '');
      helper.style.position = 'fixed';
      helper.style.left = '-9999px';
      document.body.appendChild(helper);
      helper.select();
      const ok = document.execCommand('copy');
      helper.remove();
      $('copy-status').className = ok ? 'status good' : 'status warn';
      $('copy-status').textContent = ok ? 'Standard form copied with process receipt.' : 'Copy failed. Select and copy from the standard form manually.';
      announce($('copy-status').textContent);
    }

    function showModal(id) {
      $(id).classList.add('active');
      document.body.style.overflow = 'hidden';
    }

    function closeModal(id) {
      $(id).classList.remove('active');
      if (!document.querySelector('.modal-backdrop.active')) {
        document.body.style.overflow = '';
      }
      setTimeout(() => lastFocus?.focus?.(), 0);
    }

    function showError(id, message) {
      const el = $(id);
      el.hidden = false;
      el.textContent = message;
    }

    function hideError(id) {
      const el = $(id);
      el.hidden = true;
      el.textContent = '';
    }

    function trapFocus(event) {
      if (event.key !== 'Tab') return;
      const activeModal = document.querySelector('.modal-backdrop.active .modal');
      if (!activeModal) return;
      const focusables = [...activeModal.querySelectorAll('button, input, textarea, [href], [tabindex]:not([tabindex="-1"])')]
        .filter((el) => !el.disabled && !el.hidden && el.offsetParent !== null);
      if (!focusables.length) return;
      const first = focusables[0];
      const last = focusables[focusables.length - 1];
      if (event.shiftKey && document.activeElement === first) {
        event.preventDefault();
        last.focus();
      } else if (!event.shiftKey && document.activeElement === last) {
        event.preventDefault();
        first.focus();
      }
    }

    $('prepare-btn').addEventListener('click', prepareStatements);
    $('example-btn').addEventListener('click', loadExample);
    $('clear-btn').addEventListener('click', clearAll);
    $('implicit-btn').addEventListener('click', openImplicitModal);
    $('save-implicit-btn').addEventListener('click', saveImplicit);
    $('cancel-implicit-btn').addEventListener('click', () => closeModal('implicit-backdrop'));
    $('save-support-btn').addEventListener('click', saveSupport);
    $('cancel-support-btn').addEventListener('click', () => closeModal('support-backdrop'));
    $('copy-btn').addEventListener('click', requestCopy);
    $('confirm-copy-btn').addEventListener('click', () => pendingCopy && copySubmission());
    $('cancel-copy-btn').addEventListener('click', () => {
      pendingCopy = false;
      closeModal('confirm-backdrop');
    });

    $('statement-list').addEventListener('input', (event) => {
      const action = event.target.dataset.action;
      const id = event.target.dataset.id;
      if (action === 'edit') updateText(id, event.target.value);
    });

    $('statement-list').addEventListener('click', (event) => {
      const button = event.target.closest('button[data-action]');
      if (!button) return;
      const action = button.dataset.action;
      const id = button.dataset.id;
      if (action === 'tag') tagUnit(id, button.dataset.tag);
      if (action === 'split') splitUnit(id);
      if (action === 'merge-prev') mergeUnit(id, 'prev');
      if (action === 'merge-next') mergeUnit(id, 'next');
      if (action === 'delete') deleteUnit(id);
    });

    document.addEventListener('keydown', (event) => {
      trapFocus(event);
      if (event.key !== 'Escape') return;
      if ($('implicit-backdrop').classList.contains('active')) closeModal('implicit-backdrop');
      if ($('confirm-backdrop').classList.contains('active')) closeModal('confirm-backdrop');
    });

    setInterval(updateStats, 1000);
    render();
  </script>
</body>
